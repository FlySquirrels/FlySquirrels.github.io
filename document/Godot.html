# 项目实战

平滑相机组件：Camera2D

  1. 作为角色的子节点
  2. Drag启用属性 -horizontal -vertical
  3. 平滑属性 position smoothing
  4. limit 配置上下左右可视距离 smoothed 显示极限到距离有个减速过程
   # 假设镜头已是角色子节点，并已获取镜头和地图
   # 初始化设置镜头范围
   func initCamera2DPositionArea(camera:Camera2D,tilemap:TileMap):
       # 根据地图设置镜头极限显示区域
       var used:Rect2i  = tilemap.get_used_rect();
       var title_size:Vector2i = tilemap.tile_set.tile_size;
       #camera.limit_top = used.position.y * title_size.y;
       camera.limit_bottom = used.end.y * title_size.y;
       camera.limit_right = used.end.x * title_size.x;
       camera.limit_left = used.position.x * title_size.x;
       camera.limit_smoothed = true;

       # 开启自由区域
       camera.drag_horizontal_enabled = true;
       camera.drag_vertical_enabled = true;
       # 镜头内自由自动，不跟随区域
       camera.drag_top_margin = 0.6;
       camera.drag_bottom_margin = 0.6;
       camera.drag_left_margin = 0.6;
       camera.drag_right_margin = 0.6;
       
       # 取消初始过度动画
       camera.reset_smoothing();
